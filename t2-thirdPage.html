របាំប្រពៃណីខ្មែរ
ការសំដែងរបាំគោះអង្រែ

របាំគោះអង្រែ ជារបាំប្រជាប្រិយមួយ ដែលបង្កើតឡើងដោយប្រើប្រាស់ដុំឈើ អង្រែ ហើយនិងត្បាល់ ដែលកសិករតែងតែ យកវាមកបុកកម្ទេច សំបកគ្រាប់ស្រូវចេញ។ របាំនេះ តែងតែប្រារឰធ្វើបន្ទាប់ ពីរដូវច្រូតកាត់ហើយ គឺធ្វើដើម្បីថ្លែងអំណរគុណ និងការគោរពចំពោះព្រលឹងអារក្ខអ្នកតា ទាំងឡាយដែលនាំមកនូវភាព រុងរឿងនិង ការលូតលាស់ដល់ការដាំដុះស្រូវ ។ ប្រជាកសិករនាំគ្នារាំលេងកំសាន្តនៅពេលឈប់ សម្រាកពីការច្រូតកាត់ ឬក្នុង ពីធី។របាំនេះពុំមានទំនុកច្រៀង ។

ការសំដែងរបាំនេសាទ

ពាក្យចាស់បុរាណខ្មែពីដើមបានលើកឡើងថា “ទីណាមានទឹក ទីនោះមានត្រី” ដូច្នេះហើយបានជាពួកគាត់បង្កើត របាំនេសាទ ដែលកើតចេញពីប្រជាជនរស់នៅក្នុងតំបន់បឹងទន្លេសាប ហើយប្រជាជននៅតំបន់នោះភាគច្រើនប្រកបមុខរបរនេសាទដើម្បីចិញ្ចឹមជីវិតប្រចាំថ្ងៃ ។ ក្រោយសម្រាកពីការងារ ពួកគេបានជ្រើសរើសរបាំនេសាទត្រីយកមករាំលេងកម្សាន្តក្នុងពិធីផ្សេងៗ ដូជាពិធីបំបួសនាគ និងចូលឆ្នាំខ្មែរ។

ការសំដែងរបាំគោះត្រឡោក⁣⁣

របាំ​គោះត្រឡោក ឬ​របាំ​កណ្តូបសេះ គឺជា​មរតក​របស់​ជ​ន​ជាតិខ្មែរ​តាំងពី​យូរលង់​មកហើយ​។ បើតាម​ជំនឿ ត្រឡោក​ទាំងពីរ ជា​និមិត្តរូប​នៃ​ការ​ចងសម្ព័ន្ធ​រវាង​ភេទ​ស្រី និង​ភេទ​ប្រុស ដែល​នាំមក​នូវ​ភាព​រីកចម្រើន​នៃ​ការបន្ត​ពូជ​។ នេះ​បើតាម​ឯ​សារ​វប្បធម៌ និង​អរិយធម៌​ខ្មែរ​ដែល​រៀបរៀង​ដោយ​សាស្ត្រាចារ្យ ពន់ ឆាយ​។ ​របាំ​កណ្តូបសេះនេះ ត្រូវបាន​ដាក់​ឲ្យ​សម្តែង​ជា​ផ្លូវ​ការជូន​ភ្ញៀវ​ជាតិ និង​អន្តរជាតិ​ទស្សនា​នៅក្នុង​ឆ្នាំ​១៩៦៦​ ។​

ការសំដែងរបាំត្រុដិ

របាំត្រុដិនេះ គឺជាបុណ្យឆ្លងពីឆ្នាំចាស់ចូលឆ្នាំថ្មី ពីភាពឧបទ្រពចង្រៃ ទៅភាពសម្បូរណ៍ សប្បាយផង។ ពាក្យ ត្រុដិ សំដៅដល់ពេល ដែលឆ្នាំ ចាស់ដាច់ ដើម្បីត្រាច់ផ្លាស់ចូលឆ្នាំថ្មី។ តាមអត្ថន័យនេះ របាំត្រុដិ ដែលត្រូវ បានប្រារព្ធធ្វើនោះ គឺត្រូវកំណត់ដំណើរកាត់ផ្តាច់ នៃឆ្នាំ នីមួយៗ ក្នុងទស្សនៈ របស់ជនជាតិខ្មែរ ដែលបានរស់នៅខ្ពង់ រាប និងវាលទំនាប ។

ការសំដែងរបាំបេះក្រវ៉ាញ

របាំ​បេះ​ផ្លែ​ក្រវាញ ឬហៅ​ខ្លី របាំបេះ​ក្រវាញ គឺ​ជា​របាំ​ដ៏​ល្បីល្បាញ​មួយ​ប្រភេទ​ប្រចាំ​ខេត្តពោធិ៍​សាត់ ត្រូវ​បាន​គេ​បង្ហើប​ឱ្យ​ដឹង​ថា​ ជា​ការ​រក​ឃើញ​និង​បង្កើត​ឡើង​ដោយ​លោក ម៉ៅ សោភ័ណ ដែល​ច្នៃ​ចម្លង​ចេញ​ពី​ជីវភាព​រស់នៅ​ជាក់ស្តែង​ នៃ​ជនជាតិ​ព័រ​ និង​ជនជាតិ​ជង ​តាម​តំបន់​ភ្នំ​ក្រវាញ​នៃ​ខេត្ត​ពោធិ៍សាត់។ របាំ​បេះក្រវាញ​ត្រូវ​បាន​បង្កើត​ឡើង​ កាល​ពី​សម័យ​រដ្ឋ​កម្ពុជា ។

ការសំដែងរបាំក្ងោកប៉ៃលិន

របាំ “កោ្ងកប៉ៃលិន” ជារបាំរបស់ជនជាតិកូឡាក្នុងប្រទេសខ្មែរ ដែលបានមកសាងជីវិតរស់នៅនាតំបន់ជួរភ្នំយ៉ាតនៃក្រុងប៉ៃលិន ។ របាំនេះមានការទាក់ទងយ៉ាងជ្រាលជ្រៅជាទីបំផុតទៅនឹងជំនឿរបស់ខ្មែរដើម ជាពិសេសទៅលើអារក្សអ្នកតា ។ ​ប្រជាពលរដ្ឋខ្មែរ​គោរព ជំនឿនេះ តាំងពី​មុន​ពេល​ដែល​ព្រះពុទ្ធសាសនា និង​ព្រហ្មញ្ញសាសនា​ ហូរ​ចូល​មកក្នុង​ជ្រោយ​សុវណ្ណ​ភូមិ​នោះ​ម្ល៉េះ ។​
